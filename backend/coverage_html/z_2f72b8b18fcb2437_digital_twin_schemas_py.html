147class ClinicalTextAnalysisRequest(BaseModel): 

148 """Request schema for analyzing clinical text.""" 

149 

150 text: str = Field(..., description="Clinical text to analyze") 

151 digital_twin_id: Optional[str] = Field(None, description="ID of associated digital twin") 

152 patient_id: Optional[str] = Field(None, description="ID of associated patient") 

153 analysis_type: str = Field("diagnostic_impression", description="Type of analysis to perform") 

154 detect_phi: bool = Field(True, description="Whether to detect and remove PHI") 

155 

156 @field_validator('text') 

157 def text_not_empty(cls, v): 

158 """Validate that text is not empty.""" 

159 if not v or not v.strip(): 

160 raise ValueError("Text cannot be empty") 

161 return v 

162 

163 @field_validator('analysis_type') 

164 def validate_analysis_type(cls, v): 

165 """Validate analysis type.""" 

166 valid_types = [ 

167 "diagnostic_impression", 

168 "risk_assessment", 

169 "treatment_recommendation", 

170 "clinical_insight" 

171 ] 

172 if v not in valid_types: 

173 raise ValueError(f"Analysis type must be one of: {', '.join(valid_types)}") 

174 return v
