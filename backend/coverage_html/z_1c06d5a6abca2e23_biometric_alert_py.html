15class AlertPriority(str, Enum): 

16 """Priority levels for biometric alerts.""" 

17 URGENT = "urgent" 

18 WARNING = "warning" 

19 INFORMATIONAL = "informational"

22class AlertStatus(str, Enum): 

23 """Status of a biometric alert.""" 

24 NEW = "new" 

25 ACKNOWLEDGED = "acknowledged" 

26 IN_PROGRESS = "in_progress"

31class BiometricAlert: 

32 """ 

33 BiometricAlert domain entity. 

34  

35 Represents a clinical alert generated from biometric data analysis

39 def __init__( 

40 self, 

41 patient_id: UUID, 

42 alert_type: str, 

43 description: str, 

44 priority: AlertPriority, 

45 data_points: List[Dict[str, Any]], 

46 rule_id: UUID, 

47 alert_id: Optional[UUID] = None,

49 updated_at: Optional[datetime] = None,

58 """ 

59 Initialize a BiometricAlert. 

60  

61 Args: 

62 patient_id: ID of the patient this alert is for 

63 alert_type: Type of alert (e.g., elevated_heart_rate, sleep_disruption) 

64 description: Human-readable description of the alert 

65 priority: Urgency level of the alert 

66 data_points: Biometric data points that triggered the alert 

67 rule_id: ID of the clinical rule that generated this alert 

68 alert_id: Unique identifier for this alert (generated if not provided) 

69 created_at: When the alert was created (current time if not provided) 

70 updated_at: When the alert was last updated (current time if not provided) 

71 status: Current status of the alert (NEW if not provided) 

72 acknowledged_by: ID of the provider who acknowledged the alert 

73 acknowledged_at: When the alert was acknowledged

75 resolved_at: When the alert was resolved

77 metadata: Additional contextual information

79 self.patient_id = patient_id 

80 self.alert_type = alert_type 

81 self.description = description 

82 self.priority = priority 

83 self.data_points = data_points 

84 self.rule_id = rule_id 

85 

86 # Generate ID and timestamps if not provided 

87 self.alert_id = alert_id or uuid4() 

88 self.created_at = created_at or datetime.utcnow() 

89 self.updated_at = updated_at or datetime.utcnow() 

90 

91 # Status and acknowledgment 

92 self.status = status or AlertStatus.NEW 

93 self.acknowledged_by = acknowledged_by 

94 self.acknowledged_at = acknowledged_at 

95 self.resolved_by = resolved_by

97 self.resolution_notes = resolution_notes

99 # Additional data 

100 self.metadata = metadata or {}

102 def acknowledge(self, provider_id: UUID) -> None: 

103 """ 

104 Acknowledge the alert. 

105  

106 Args: 

107 provider_id: ID of the provider acknowledging the alert 

108 """ 

109 if self.status == AlertStatus.NEW:

115 def mark_in_progress(self, provider_id: UUID) -> None: 

116 """ 

117 Mark the alert as in progress. 

118  

119 Args: 

120 provider_id: ID of the provider working on the alert 

121 """ 

122 if self.status in [AlertStatus.NEW, AlertStatus.ACKNOWLEDGED]: 

123 self.status = AlertStatus.IN_PROGRESS 

124 

125 # If not already acknowledged, acknowledge it 

126 if not self.acknowledged_by: 

127 self.acknowledged_by = provider_id 

128 self.acknowledged_at = datetime.utcnow() 

129 

130 self.updated_at = datetime.utcnow()

132 def resolve(self, provider_id: UUID, notes: Optional[str] = None) -> None: 

133 """ 

134 Resolve the alert.

136 Args: 

137 provider_id: ID of the provider resolving the alert 

138 notes: Optional notes about the resolution

140 if self.status in [AlertStatus.NEW, AlertStatus.ACKNOWLEDGED, AlertStatus.IN_PROGRESS]: 

141 self.status = AlertStatus.RESOLVED 

142 self.resolved_by = provider_id 

143 self.resolved_at = datetime.utcnow() 

144 

145 if notes: 

146 self.resolution_notes = notes 

147 

148 # If not already acknowledged, acknowledge it 

149 if not self.acknowledged_by: 

150 self.acknowledged_by = provider_id 

151 self.acknowledged_at = datetime.utcnow() 

152 

153 self.updated_at = datetime.utcnow() 

154

156 """ 

157 Dismiss the alert as not requiring action.

163 if self.status in [AlertStatus.NEW, AlertStatus.ACKNOWLEDGED, AlertStatus.IN_PROGRESS]: 

164 self.status = AlertStatus.DISMISSED 

165 self.resolved_by = provider_id
